29  template<class V>
30  V ls_solve(const std::vector<V>& A, const std::vector<typename V::ScalarType>& b)
31  {
32  typename VecT_to_MatT<V>::MatT ATA(0);
33  V ATb(0);
34  for(int n = 0; n < A.size(); ++n) {
35  ATA += outer_product(A[n], A[n]);
36  ATb += b[n]*A[n];
37  }
38  return invert(ATA)*ATb;
39  }
