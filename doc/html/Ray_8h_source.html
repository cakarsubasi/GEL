100  void refract(const CGLA::Vec3f& normal, float new_ior)
101  {
102  float ref_ratio = ior/new_ior;
103  float cos_N_I = dot(normal, direction);
104  CGLA::Vec3f norm(normal);
105 
106  if(cos_N_I > 0.0f)
107  {
108  norm = -norm;
109  cos_N_I = dot(norm, direction);
110  }
111 
112  float selector = 1+(ref_ratio*ref_ratio)*(cos_N_I*cos_N_I - 1);
113 
114  if(selector > 0.0f)
115  {
116  direction = norm*(ref_ratio*(-cos_N_I) - sqrt(selector))
117  + direction*ref_ratio;
118  ior = new_ior;
119  }
120  else
121  // Total internal reflection.
122  reflect(normal);
123  }
