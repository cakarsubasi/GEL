29  XForm(const CGLA::Vec3d& _llf, const CGLA::Vec3d& _urt, const CGLA::Vec3i& _DIM, double _margin=0.1):
30  llf(_llf), urt(_urt), DIM(_DIM)
31  {
32  if(urt[0]<llf[0])
33  {
34  margin = 0.0;
35  scale = 1;
36  llf = CGLA::Vec3d(0);
37  urt = CGLA::Vec3d(DIM);
38  }
39  else
40  {
41  CGLA::Vec3d d = urt - llf;
42  margin = d.max_coord() * _margin;
43  double scale_x = (DIM[0])/(d[0] + 2.0 * margin);
44  double scale_y = (DIM[1])/(d[1] + 2.0 * margin);
45  double scale_z = (DIM[2])/(d[2] + 2.0 * margin);
46  scale = fmin(fmin(scale_x, scale_y), scale_z);
47  }
48  }
