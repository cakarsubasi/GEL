114 /* + m[2][0] * (m[0][1]*(m[1][2]*m[3][3]-m[3][2]*m[1][3])- */
115 /* m[1][1]*(m[0][2]*m[3][3]-m[3][2]*m[0][3])+ */
116 /* m[3][1]*(m[0][2]*m[1][3]-m[1][2]*m[0][3])) */
117 /* - m[3][0] * (m[0][1]*(m[1][2]*m[2][3]-m[2][2]*m[1][3])- */
118 /* m[1][1]*(m[0][2]*m[2][3]-m[2][2]*m[0][3])+ */
119 /* m[2][1]*(m[0][2]*m[1][3]-m[1][2]*m[0][3])); */
120  typedef typename M::ScalarType ScalarType;
121  ScalarType a1, a2, a3, a4, b1, b2, b3, b4, c1, c2, c3, c4, d1, d2, d3, d4;

123  /* assign to individual variable names to aid selecting */
124  /* correct elements */
125 
126  a1 = m[0][0]; b1 = m[1][0]; c1 = m[2][0]; d1 = m[3][0];
127  a2 = m[0][1]; b2 = m[1][1]; c2 = m[2][1]; d2 = m[3][1];
128  a3 = m[0][2]; b3 = m[1][2]; c3 = m[2][2]; d3 = m[3][2];
