100  EdgeID find_edge(NodeID n0, NodeID n1) const
101  {
102  if(valid_node_id(n0) && valid_node_id(n1)) {
103  auto it = edge_map[n0].find(n1);

113  {
114  if(valid_node_id(n0) && valid_node_id(n1) &&
115  find_edge(n0, n1) == InvalidEdgeID) {
116  size_t id = no_edges_created++;
117  edge_map[n0][n1] = id;
118  edge_map[n1][n0] = id;

243  unsigned i=0;

245  for(NodeID n: node_ids())
