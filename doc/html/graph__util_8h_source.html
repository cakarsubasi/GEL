106  class LineSegment {
107  const CGLA::Vec3d p0,p1;
108  const CGLA::Vec3d dir;
109  const double sqlen;
110  public:
111  LineSegment(const CGLA::Vec3d& _p0, const CGLA::Vec3d& _p1): p0(_p0), p1(_p1), dir(p1-p0), sqlen(sqr_length(dir)) {}
112 
113  LineProj sqr_distance(const CGLA::Vec3d& p) const {
114  CGLA::Vec3d v = p - p0;
115  double proj = dot(v, dir)/sqlen;
116  if(proj < 0)
117  return {sqr_length(p-p0),0.0};
118  if(proj >1)
119  return {sqr_length(p-p1),1.0};
120  return { sqr_length(p0 + (p1-p0) * proj - p), proj};
121  }
122  };
